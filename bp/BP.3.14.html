Program P1; var a, c, s, h, n:real; k:char; begin writeln('Alegeti optiunea: 1-cateta, 2-ipotenuza, 3-inaltimea pe ipotenuza, 4-aria'); write('Optiunea: '); readln(k); write('Introduceti latura: '); readln(n); case k of '1': a:=n; '2': a:=sqrt(sqr(n)/2); '3': a:=sqrt(sqr(n)/2)/2; '4': a:=2*sqrt(n); else write('Aceasta optiune nu exista'); end; c:=sqrt(sqr(a)*2); h:=c/2; s:=sqr(a)/2; writeln('In triunghiul dat cateta este ',a:0:2,' ipotenuza este ',c:0:2,' inaltimea pe ipotenuza ',h:0:2,' aria ',s:0:2); end.
